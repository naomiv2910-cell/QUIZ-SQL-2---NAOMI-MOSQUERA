SELECT sede_codigo AS codigo_sede,
    periodo_anio AS anio_reporte,
    actividad_codigo AS cod_actividad,
    actividad_nombre AS nombre_actividad
FROM tic2022 ORDER BY SEDE_CODIGO ASC
LIMIT 50;

CREATE TABLE tic_sedes_resumen (
    resumen_id INT PRIMARY KEY,
    sede_codigo INT NOT NULL,
    anio INTNOT NULL,
    total_actividades INT NOT NULL DEFAULT 0,
    tiene_internet BOOLEAN NOT NULL,
    fecha_carga DATETIME NOT NULL,
    UNIQUE(sede_codigo, anio)
);

#PUNTO 3 (VIEW)

SELECT * FROM tic_sedes_resumen 

#PUNTO 4
 SELECT 
    SUBSTR(sede_codigo, 1, 2) AS cod_departamento,
    COUNT(DISTINCT sede_codigo) AS total_sedes_unicas
FROM tic2023
WHERE COALESCE(actividad_codigo, '05') = '05'
GROUP BY cod_departamento
HAVING total_sedes_unicas > 500
ORDER BY total_sedes_unicas DESC;

#PUNTO 5


SELECT 
    t22.sede_codigo AS codigo_sede,
    t22.total_2022 AS total_act_2022,
    t23.total_2023 AS total_act_2023,
    (t23.total_2023 - t22.total_2022) AS diferencia,
    CASE 
        WHEN (t23.total_2023 - t22.total_2022) > 0 THEN 'creció'
        WHEN (t23.total_2023 - t22.total_2022) < 0 THEN 'decreció'
        ELSE 'sin cambio'
    END AS tendencia
FROM (
    SELECT sede_codigo, COUNT(*) AS total_2022
    FROM tic2022
    GROUP BY sede_codigo
    HAVING COUNT(*) >= 2  
) t22
INNER JOIN (
    SELECT sede_codigo, COUNT(*) AS total_2023
    FROM tic2023
    GROUP BY sede_codigo
) t23 ON t22.sede_codigo = t23.sede_codigo
ORDER BY diferencia DESC
LIMIT 30;
